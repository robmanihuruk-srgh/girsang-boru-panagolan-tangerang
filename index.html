// ==================== MODIFIKASI FUNGSI SISTEM (BACKUP, RESTORE, REFRESH) ==================== // 1. Fungsi Backup Data: Mengunduh seluruh database Firebase ke file JSON async function backupData() { try { showToast('info', 'Backup', 'Sedang menyiapkan data...'); // Ambil semua data utama dari variabel global yang sudah tersinkron dengan Firebase const fullBackup = { metadata: { appName: "Girsang Boru Panagolan Tangerang - Management System", version: "2.0", backupDate: new Date().toISOString(), backupBy: currentUser.username }, members: members, attendance: attendance, finance: finance, users: users }; // Konversi ke string JSON const dataStr = JSON.stringify(fullBackup, null, 2); const dataUri = 'data:application/json;charset=utf-8,' + encodeURIComponent(dataStr); // Buat nama file dengan timestamp const timestamp = new Date().toISOString().replace(/[:.]/g, '-').slice(0, 19); const exportFileDefaultName = `Backup_Girsang_System_${timestamp}.json`; // Trigger download const linkElement = document.createElement('a'); linkElement.setAttribute('href', dataUri); linkElement.setAttribute('download', exportFileDefaultName); linkElement.click(); showToast('success', 'Backup Berhasil', `File ${exportFileDefaultName} telah diunduh.`); } catch (error) { console.error("Backup Error:", error); showToast('error', 'Backup Gagal', 'Terjadi kesalahan saat memproses backup.'); } } // 2. Fungsi Restore Data: Membaca file JSON dan menuliskannya kembali ke Firebase function showRestoreDialog() { // Buat input file secara dinamis const input = document.createElement('input'); input.type = 'file'; input.accept = '.json'; input.onchange = e => { const file = e.target.files[0]; if (!file) return; const reader = new FileReader(); reader.readAsText(file, 'UTF-8'); reader.onload = async readerEvent => { try { const content = JSON.parse(readerEvent.target.result); // Validasi struktur file backup if (!content.members || !content.attendance || !content.finance) { throw new Error("Format file backup tidak valid."); } if (!confirm(`⚠️ PERINGATAN: Restore akan menimpa data saat ini dengan data dari cadangan tanggal ${new Date(content.metadata.backupDate).toLocaleString()}. Lanjutkan?`)) { return; } showToast('info', 'Restore', 'Sedang memproses data ke Firebase...'); // Restore setiap node ke Firebase await db.ref('members').set(mapArrayToFirebase(content.members)); await db.ref('attendance').set(mapArrayToFirebase(content.attendance)); await db.ref('finance').set(mapArrayToFirebase(content.finance)); // Khusus users, opsional jika ingin direstore juga (hati-hati bisa merubah akses login saat ini) if (content.users && confirm("Apakah Anda juga ingin merestore data Pengguna/Admin?")) { await db.ref('users').set(mapArrayToFirebase(content.users)); } showToast('success', 'Restore Berhasil', 'Database telah diperbarui. Aplikasi akan memuat ulang.'); setTimeout(() => location.reload(), 2000); } catch (error) { console.error("Restore Error:", error); showToast('error', 'Restore Gagal', 'File rusak atau format tidak sesuai: ' + error.message); } }; }; input.click(); } // Helper untuk membersihkan data sebelum dikirim balik ke Firebase (menghapus firebaseKey agar tidak duplikat di dalam objek) function mapArrayToFirebase(array) { const obj = {}; array.forEach(item => { const key = item.firebaseKey || db.ref().push().key; const cleanItem = { ...item }; delete cleanItem.firebaseKey; // Hapus key agar struktur Firebase bersih obj[key] = cleanItem; }); return obj; } // 3. Fungsi Refresh Info: Memperbarui tampilan info sistem dan memuat ulang data dari Firebase async function showSystemInfo() { const systemInfoContent = document.getElementById('systemInfoContent'); // Tampilkan status loading systemInfoContent.innerHTML = `

🔄 Sedang menyegarkan data...

`; // Paksa load ulang data dari Firebase await loadAllData(); const now = new Date(); const uptime = Math.floor((now - window.appStartTime) / 60000); // dalam menit systemInfoContent.innerHTML = `
Versi Aplikasi
2.0.0-Stable
User Aktif
${currentUser.username} (${currentUser.role})
Total Database
${members.length} Anggota
Uptime Sesi
${uptime} Menit

📍 Lokasi: Tangerang

🕒 Last Sync: ${now.toLocaleTimeString()}

🗄️ Firebase: Connected (girsangborupanagolantangerang)

`; }
